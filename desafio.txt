//posicao dos vencedores
SELECT codpiloto, nVolta, tempoVolta
FROM teste_php.corrida
where nVolta = 4
order by tempoVolta asc

//qtd de voltas que cada piloto deu
SELECT  
            codPiloto, 
            piloto, 
            COUNT(nVolta) AS qtd_voltas 
        FROM teste_php.corrida
        GROUP BY codPiloto, piloto
        HAVING COUNT(nVolta) > 1;
